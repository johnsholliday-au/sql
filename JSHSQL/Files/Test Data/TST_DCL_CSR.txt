DECLARE CURSOR-5 CURSOR FOR 
SELECT T1 . NAME , T1 . UNIQUERULE , T1 . CLUSTERING , T1 . CLUSTERED , T1 . CLUSTERRATIO 
    , T1 . FIRSTKEYCARD , T1 . FULLKEYCARD , T1 . NLEAF , T1 . NLEVELS , T2 . COLNAME 
    , T2 . COLSEQ , T2 . ORDERING 
FROM SYSINDEXES T1 
     , SYSKEYS T2 
WHERE T1 . TBCREATOR = : H 
AND T1 . TBNAME = : H 
AND T2 . IXCREATOR = T1 . CREATOR 
AND T2 . IXNAME = T1 . NAME 
ORDER BY T1 . NAME , T2 . COLSEQ
